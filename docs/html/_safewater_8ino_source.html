537 * and refresh it with default values.
538 *
539 * Returns True if initialisation was required.

542{
543
544 int iEEPROMCheck1;
545 int iEEPROMCheck2;
546
547 bool bInitialiseEEPROM = false;
548
549 // Check to see if the EEPROM has been initialised
550 iEEPROMCheck1 = EEPROM.readByte(EEPROM_ADDRESS_CHECK1);
551 iEEPROMCheck2 = EEPROM.readByte(EEPROM_ADDRESS_CHECK2);
552
553 Trace(F("EEPROM PreCheck1 = "));
554 Traceln(iEEPROMCheck1);
555 Trace(F("EEPROM PreCheck2 = "));
556 Traceln(iEEPROMCheck2);
557
558 if ((iEEPROMCheck1 != EEPROM_CHECKBYTE1_VALUE) || (iEEPROMCheck2 != EEPROM_CHECKBYTE2_VALUE))
559 {
560
561 Traceln(F("EEPROM Init Required."));
562
563 // Need to initialise EEPROM before reading
564 InitialiseEEPROMValues();
565
566 bInitialiseEEPROM = true;
567 }
568 else
569 {
570 Traceln(F("EEPROM Init NOT required."));
571
572 bInitialiseEEPROM = false;
573 }
574
575
576 // Load Settings
577 ControllerConfigSettings.LoadFromEEPROM();
578
579 return bInitialiseEEPROM;

730void setup()
731{
732
733 // Activate Watchdog Timer
734 wdt_enable(WDTO_8S);
735
736 // Serial Port Setup
737 beginDebug();
738
739 Traceln(F("Begin Startup..."));
740 Traceln(SYSTEM_VERSION);
741
742 Trace(F("Startup FreeRAM:"));
743 Traceln(Utilities.freeRam());
744
745 // Set I/O
746 InitialiseIO();
747 //WaterPump.switchPumpNegativeLogic();
748 digitalWrite(UVC_OUT_IO, UVC_LAMP_OFF);
749 digitalWrite(PUMP_OUT_IO, WaterPump.iPumpOffState);
750
751 // Initialise Alarm Buzzer
752 setPwmFrequency(9, ALARM_BUZZER_FREQUENCY);
753
754 // Initialise EEPROM variables
755 EEPROM.setMemPool(0, EEPROMSizeATmega328);
756 EEPROM.setMaxAllowedWrites(EEPROM_MAX_WRITES);
757
758 ReadEEPROMValues();
759
760
761 boolean bValidRuntimeReadingFound = false;
762 unsigned int uiCheckCount = 0;
763
764 while ((bValidRuntimeReadingFound == false) && (uiCheckCount <= 10))
765 {
766
767 // Check the Max Runtime Jumper
768 if (digitalRead(MAX_RUNTIME_JUMPER_IN_IO) == false)
769 {
770 delay(50);
771 if (digitalRead(MAX_RUNTIME_JUMPER_IN_IO) == false)
772 {
773 uiMaxRuntimeSetting = MAX_RUNTIME_MEXICO;
774 bValidRuntimeReadingFound = true;
775 break;
776 }
777 }
778 else
779 {
780 delay(50);
781 if (digitalRead(MAX_RUNTIME_JUMPER_IN_IO) == true)
782 {
783 uiMaxRuntimeSetting = MAX_RUNTIME_COLOMBIA;
784 bValidRuntimeReadingFound = true;
785 break;
786 }
787 }
788
789 uiCheckCount++;
790
791 }
792
793 // Ensure that a default jumper value is set, if for some reason a consistent value isn't found (e.g. N/C floating jumper)
794 if (bValidRuntimeReadingFound == false)
795 {
796 uiMaxRuntimeSetting = MAX_RUNTIME_MEXICO;
797 }
798
799 // Reset Watchdog Timer
800 wdt_reset();
801
802 // Startup Delay
803 delay(2000);
804
805 // RTCC
806 // Start internal system clock running - only used for relative rather than absolute time, so exact date/time setting not important.
807 setTime(0, 0, 0, 1, 1, 2020);
808
809 // Reset Watchdog Timer
810 wdt_reset();
811
812 // Initial display for Light Indicators
813 InitialiseLights(true);
814 delay(3000);
815
816 // Deactivate Light Indicators after delay
817 InitialiseLights(false);
818
819 // Set Pump Relay Switching Delay
820 WaterPump.iPumpRelaySwitchingDelay = LOCAL_PUMP_RELAY_SWITCHING_DELAY;
821
822 // Disable Pump Ventilation
823 WaterPump.bEnablePumpVentilation = false;
824
825 Trace(F("FreeRAM after Initialisation:"));
826 Traceln(Utilities.freeRam());
827
828 Traceln(F("Startup Complete."));
829
830}
